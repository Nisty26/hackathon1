API Example 
import requests

# Analyze all images
with open('selfie.jpg', 'rb') as s, \
     open('outfit1.jpg', 'rb') as o1, \
     open('outfit2.jpg', 'rb') as o2:
    
    response = requests.post(
        'http://api.chromastyle.com/analyze/all',
        files={
            'selfie': s,
            'outfit1': o1,
            'outfit2': o2
        }
    )
    
# Get complete analysis
result = response.json()
print(result['skin']['undertone'])  # Warm, Cool, or Neutral
print(result['outfits']['harmony'])  # Complementary, Analogous, or Clashing
